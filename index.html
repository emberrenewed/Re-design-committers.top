  3. 03

# set a GitHub token with read:user and read:org
export GITHUB_TOKEN="paste-your-token-here"

# rank the top 50 worldwide contributors
go run . --preset worldwide --amount 50 --consider 250

# save a CSV report
go run . --preset worldwide --amount 500 --output csv --file output.csv
